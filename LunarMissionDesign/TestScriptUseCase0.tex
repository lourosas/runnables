\textbf{Use Case 0 Test Sequence}
\end{center}
\noindent
\begin{enumerate}
   \item If in simulation, load the Simulation File
   \begin{enumerate}
      \item Currently, this is called the ``Feeder File"
   \end{enumerate}
      \begin{enumerate}
         \item Load the Simulation File for the Rocket
         \item Load the Simulation File for the Support Structure
         \begin{enumerate}
            \item Load the Simultion File for the (Platform) Supports
         \end{enumerate}
         \item Load Simulation File for the Capsule
      \end{enumerate}
   \item Load in the Initialization Data
   \begin{enumerate}
      \item Load in the Initialization Data for the Rocket
      \item Load in the Initialization Data for the Support Structure
         \begin{enumerate}
            \item Load in the Initialization Data for the (Platform)
            Supports Data
         \end{enumerate}
      \item Load in the Initialization Data for the Capsule
   \end{enumerate}
   \item Monitor Initialization State
   \begin{enumerate}
      \item Monitor Rocket Initialization State
      \item Monitor Support Structure Initialization State
      \begin{enumerate}
         \item Monitor (Platform) Supports Initialization State
      \end{enumerate}
      \item Monitor Capsule Initialization State
   \end{enumerate}
   \item Display Initialization State
   \begin{enumerate}
      \item Display Rocket Initialization State
      \item Display Support Structure Initialization State
      \item Display (Platform) Supports Initialization State
      \item Display Capsule Initialization State
   \end{enumerate}
\end{enumerate}
